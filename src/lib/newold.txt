export const DB = {
  getCustomers: () => {
    if (typeof window === 'undefined') return []
    return JSON.parse(localStorage.getItem('customers') || '[]')
  },
  saveCustomers: (data: any) => {
    localStorage.setItem('customers', JSON.stringify(data))
  },
